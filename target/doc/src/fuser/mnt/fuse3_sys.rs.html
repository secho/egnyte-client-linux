1//! Native FFI bindings to libfuse3.
2//!
3//! This is a small set of bindings that are required to mount/unmount FUSE filesystems and
4//! open/close a fd to the FUSE kernel driver.
5
6#![warn(missing_debug_implementations)]
7#![allow(missing_docs)]
8
9use super::fuse2_sys::fuse_args;
10use libc::c_void;
11use libc::{c_char, c_int};
12
13extern "C" {
14    // Really this returns *fuse_session, but we don't need to access its fields
15    pub fn fuse_session_new(
16        args: *const fuse_args,
17        op: *const c_void, // This argument is really a *const fuse_lowlevel_ops, but we don't use them
18        op_size: libc::size_t,
19        userdata: *mut c_void,
20    ) -> *mut c_void;
21    pub fn fuse_session_mount(
22        se: *mut c_void, // This argument is really a *fuse_session
23        mountpoint: *const c_char,
24    ) -> c_int;
25    // This function's argument is really a *fuse_session
26    pub fn fuse_session_fd(se: *mut c_void) -> c_int;
27    // This function's argument is really a *fuse_session
28    pub fn fuse_session_unmount(se: *mut c_void);
29    // This function's argument is really a *fuse_session
30    pub fn fuse_session_destroy(se: *mut c_void);
31}
