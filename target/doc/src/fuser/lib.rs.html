71const fn default_init_flags(#[allow(unused_variables)] capabilities: u32) -> u32 {
72    #[cfg(not(feature = "abi-7-28"))]
73    {
74        INIT_FLAGS
75    }
76
77    #[cfg(feature = "abi-7-28")]
78    {
79        let mut flags = INIT_FLAGS;
80        if capabilities & FUSE_MAX_PAGES != 0 {
81            flags |= FUSE_MAX_PAGES;
82        }
83        flags
84    }
85}

143/// Configuration of the fuse kernel module connection
144#[derive(Debug)]
145pub struct KernelConfig {
146    capabilities: u32,
147    requested: u32,
148    max_readahead: u32,
149    max_max_readahead: u32,
150    #[cfg(feature = "abi-7-13")]
151    max_background: u16,
152    #[cfg(feature = "abi-7-13")]
153    congestion_threshold: Option<u16>,
154    max_write: u32,
155    #[cfg(feature = "abi-7-23")]
156    time_gran: Duration,
157}
158
159impl KernelConfig {
160    fn new(capabilities: u32, max_readahead: u32) -> Self {
161        Self {
162            capabilities,
163            requested: default_init_flags(capabilities),
164            max_readahead,
165            max_max_readahead: max_readahead,
166            #[cfg(feature = "abi-7-13")]
167            max_background: 16,
168            #[cfg(feature = "abi-7-13")]
169            congestion_threshold: None,
170            // use a max write size that fits into the session's buffer
171            max_write: MAX_WRITE_SIZE as u32,
172            // 1ns means nano-second granularity.
173            #[cfg(feature = "abi-7-23")]
174            time_gran: Duration::new(0, 1),
175        }
176    }
177
178    /// Set the timestamp granularity
179    ///
180    /// Must be a power of 10 nanoseconds. i.e. 1s, 0.1s, 0.01s, 1ms, 0.1ms...etc
181    ///
182    /// On success returns the previous value. On error returns the nearest value which will succeed
183    #[cfg(feature = "abi-7-23")]
184    pub fn set_time_granularity(&mut self, value: Duration) -> Result<Duration, Duration> {
185        if value.as_nanos() == 0 {
186            return Err(Duration::new(0, 1));
187        }
188        if value.as_secs() > 1 || (value.as_secs() == 1 && value.subsec_nanos() > 0) {
189            return Err(Duration::new(1, 0));
190        }
191        let mut power_of_10 = 1;
192        while power_of_10 < value.as_nanos() {
193            if value.as_nanos() < power_of_10 * 10 {
194                // value must not be a power of ten, since power_of_10 < value < power_of_10 * 10
195                return Err(Duration::new(0, power_of_10 as u32));
196            }
197            power_of_10 *= 10;
198        }
199        let previous = self.time_gran;
200        self.time_gran = value;
201        Ok(previous)
202    }
203
204    /// Set the maximum write size for a single request
205    ///
206    /// On success returns the previous value. On error returns the nearest value which will succeed
207    pub fn set_max_write(&mut self, value: u32) -> Result<u32, u32> {
208        if value == 0 {
209            return Err(1);
210        }
211        if value > MAX_WRITE_SIZE as u32 {
212            return Err(MAX_WRITE_SIZE as u32);
213        }
214        let previous = self.max_write;
215        self.max_write = value;
216        Ok(previous)
217    }
218
219    /// Set the maximum readahead size
220    ///
221    /// On success returns the previous value. On error returns the nearest value which will succeed
222    pub fn set_max_readahead(&mut self, value: u32) -> Result<u32, u32> {
223        if value == 0 {
224            return Err(1);
225        }
226        if value > self.max_max_readahead {
227            return Err(self.max_max_readahead);
228        }
229        let previous = self.max_readahead;
230        self.max_readahead = value;
231        Ok(previous)
232    }
233
234    /// Add a set of capabilities.
235    ///
236    /// On success returns Ok, else return bits of capabilities not supported when capabilities you provided are not all supported by kernel.
237    pub fn add_capabilities(&mut self, capabilities_to_add: u32) -> Result<(), u32> {
238        if capabilities_to_add & self.capabilities != capabilities_to_add {
239            return Err(capabilities_to_add - (capabilities_to_add & self.capabilities));
240        }
241        self.requested |= capabilities_to_add;
242        Ok(())
243    }
244
245    /// Set the maximum number of pending background requests. Such as readahead requests.
246    ///
247    /// On success returns the previous value. On error returns the nearest value which will succeed
248    #[cfg(feature = "abi-7-13")]
249    pub fn set_max_background(&mut self, value: u16) -> Result<u16, u16> {
250        if value == 0 {
251            return Err(1);
252        }
253        let previous = self.max_background;
254        self.max_background = value;
255        Ok(previous)
256    }
257
258    /// Set the threshold of background requests at which the kernel will consider the filesystem
259    /// request queue congested. (it may then switch to sleeping instead of spin-waiting, for example)
260    ///
261    /// On success returns the previous value. On error returns the nearest value which will succeed
262    #[cfg(feature = "abi-7-13")]
263    pub fn set_congestion_threshold(&mut self, value: u16) -> Result<u16, u16> {
264        if value == 0 {
265            return Err(1);
266        }
267        let previous = self.congestion_threshold();
268        self.congestion_threshold = Some(value);
269        Ok(previous)
270    }
271
272    #[cfg(feature = "abi-7-13")]
273    fn congestion_threshold(&self) -> u16 {
274        match self.congestion_threshold {
275            // Default to a threshold of 3/4 of the max background threads
276            None => (self.max_background as u32 * 3 / 4) as u16,
277            Some(value) => min(value, self.max_background),
278        }
279    }
280
281    #[cfg(feature = "abi-7-28")]
282    fn max_pages(&self) -> u16 {
283        ((max(self.max_write, self.max_readahead) - 1) / page_size::get() as u32) as u16 + 1
284    }
285}

1007/// Mount the given filesystem to the given mountpoint. This function will
1008/// not return until the filesystem is unmounted.
1009///
1010/// NOTE: This will eventually replace mount(), once the API is stable
1011pub fn mount2<FS: Filesystem, P: AsRef<Path>>(
1012    filesystem: FS,
1013    mountpoint: P,
1014    options: &[MountOption],
1015) -> io::Result<()> {

1017    Session::new(filesystem, mountpoint.as_ref(), options).and_then(|mut se| se.run())
1018}

1020/// Mount the given filesystem to the given mountpoint. This function spawns
1021/// a background thread to handle filesystem operations while being mounted
1022/// and therefore returns immediately. The returned handle should be stored
1023/// to reference the mounted filesystem. If it's dropped, the filesystem will
1024/// be unmounted.
1025#[deprecated(note = "use spawn_mount2() instead")]
1026pub fn spawn_mount<'a, FS: Filesystem + Send + 'static + 'a, P: AsRef<Path>>(
1027    filesystem: FS,
1028    mountpoint: P,
1029    options: &[&OsStr],
1030) -> io::Result<BackgroundSession> {
1031    let options: Option<Vec<_>> = options
1032        .iter()
1033        .map(|x| Some(MountOption::from_str(x.to_str()?)))
1034        .collect();
1035    let options = options.ok_or(ErrorKind::InvalidData)?;
1036    Session::new(filesystem, mountpoint.as_ref(), options.as_ref()).and_then(|se| se.spawn())
1037}
