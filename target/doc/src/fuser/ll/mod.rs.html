27macro_rules! errno {
28    ($x: expr) => {
29        Errno(unsafe {
30            // This is a static assertion that the constant $x is > 0
31            const _X: [(); 0 - !{
32                const ASSERT: bool = ($x > 0);
33                ASSERT
34            } as usize] = [];
35            // Which makes this safe
36            NonZeroI32::new_unchecked($x)
37        })
38    };
39}

234impl From<std::io::Error> for Errno {
235    fn from(err: std::io::Error) -> Self {
236        let errno = err.raw_os_error().unwrap_or(0);
237        match errno.try_into() {
238            Ok(i) => Errno(i),
239            Err(_) => Errno::EIO,
240        }
241    }
242}
243impl From<std::io::ErrorKind> for Errno {
244    fn from(x: std::io::ErrorKind) -> Self {
245        let err: std::io::Error = x.into();
246        err.into()
247    }
248}
249impl From<Errno> for i32 {
250    fn from(x: Errno) -> Self {
251        x.0.into()
252    }
253}

255/// A newtype for generation numbers
256///
257/// If the file system will be exported over NFS, the (ino, generation) pairs
258/// need to be unique over the file system's lifetime (rather than just the
259/// mount time). So if the file system reuses an inode after it has been
260/// deleted, it must assign a new, previously unused generation number to the
261/// inode at the same time.
262#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
263pub struct Generation(pub u64);
264impl From<Generation> for u64 {
265    fn from(fh: Generation) -> Self {
266        fh.0
267    }
268}
