28/// Unique ID for a request from the kernel
29///
30/// The FUSE kernel driver assigns a unique id to every concurrent request. This allows to
31/// distinguish between multiple concurrent requests. The unique id of a request may be
32/// reused in later requests after it has completed.
33///
34/// This can be retrieve for any request using [Request::unique].  The kernel
35/// will send an [Interrupt] request to cancel requests in progress.  It's
36/// important to handle this for any requests that may block indefinitely, like
37/// [SetLkW].
38#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
39#[cfg_attr(feature = "serializable", derive(Serialize, Deserialize))]
40pub struct RequestId(pub u64);
41impl From<RequestId> for u64 {
42    fn from(fh: RequestId) -> Self {
43        fh.0
44    }
45}

47/// A newtype for inode numbers
48///
49/// These are generated by the filesystem implementation and returned to the
50/// kernel in response to a call to [Lookup], [Create], [MkNod], [MkDir] or
51/// [SymLink].  The kernel will then pass these numbers back to the filesystem
52/// implementation when it needs to refer to a given file.  Every request has
53/// an associated [INodeNo], accessible as [Request::nodeid].
54///
55/// Reference Counting
56/// ------------------
57///
58/// Every time the kernel receives a given inode number in a response to a
59/// [Lookup], [Create], [MkNod], [MkDir] or [SymLink] request it increments an
60/// internal counter for that inode.  The filesystem implementation should do
61/// the same.  When the kernel is no longer interested in this inode it will
62/// send a [Forget] message with that counter.  The filesystem implementation
63/// should decrement its own counter and if it reaches 0 then the inode number
64/// may be recycled and your filesystem implementation may clean up its
65/// internal data-structures relating to that inode.
66///
67/// We implement conversion from [INodeNo] to [u64] but not vice-versa because
68/// not all [u64]s are valid [INodeNo]s, but the reverse is true.  So to produce
69/// a [INodeNo] from a [u64] we must be explicit.
70#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
71#[cfg_attr(feature = "serializable", derive(Serialize, Deserialize))]
72pub struct INodeNo(pub u64);
73impl From<INodeNo> for u64 {
74    fn from(fh: INodeNo) -> Self {
75        fh.0
76    }
77}

79/// A newtype for file handles
80///
81/// This corresponds to a single file description in a client program.  These
82/// are generated by the filesystem implementation in replies to [Open],
83/// [OpenDir] and [Create] requests.  It's used as a correlation id across
84/// [Read], [Write], [FSync], [IoCtl], [Poll], [FAllocate], [ReadDir],
85/// [FSyncDir], [GetLk], [SetLk], [SetLkW], [ReadDirPlus], [Lseek] and
86/// [CopyFileRange] requests.
87///
88/// A filesystem implementation may store arbitrary data as the [FileHandle], as
89/// long as it fits into 64-bits and doesn't need to change for over the lifetime
90/// of the [FileHandle].  Typically this might consist of an index into an array
91/// of [FileHandle]s that the filesystem implementation maintains.
92///
93/// Filesystems may instead implement stateless file I/O and use `0` as the
94/// [FileHandle] - although this makes it impossible to correctly implement
95/// resumable [ReadDir] in the presence of mutable directories (see [OpenDir]).
96///
97/// Lifecycle
98/// ---------
99///
100/// A [FileHandle] is owned by one or more file-descriptors (or memory
101/// mappings) in the client program.  Multiple file descriptors can point to
102/// the same [FileHandle], just as a single INode can have multiple
103/// [FileHandle]s open at one time.  Every time a single file-descriptor is
104/// closed a [Flush] request is made.  This gives filesystem implementations
105/// an opportunity to return an error message from that `close()` call.  After
106/// all the file-descriptors are closed that own a given [FileHandle] the
107/// [Release]/[ReleaseDir] request will be made.  This is an opportunity for
108/// the filesystem implementation to free any internal per-FileHandle data
109/// structures it has allocated.
110///
111/// We implement conversion from FileHandle to u64 but not vice-versa because
112/// not all u64s are valid FileHandles, but the reverse is true.  So to produce
113/// a FileHandle from a u64 we must be explicit.
114#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
115#[cfg_attr(feature = "serializable", derive(Serialize, Deserialize))]
116pub struct FileHandle(pub u64);
117
118impl From<FileHandle> for u64 {
119    fn from(fh: FileHandle) -> Self {
120        fh.0
121    }
122}

209pub trait Request: Sized {
210    /// Returns the unique identifier of this request.
211    ///
212    /// The FUSE kernel driver assigns a unique id to every concurrent request. This allows to
213    /// distinguish between multiple concurrent requests. The unique id of a request may be
214    /// reused in later requests after it has completed.
215    fn unique(&self) -> RequestId;
216
217    /// Returns the node id of the inode this request is targeted to.
218    fn nodeid(&self) -> INodeNo;
219
220    /// Returns the UID that the process that triggered this request runs under.
221    fn uid(&self) -> u32;
222
223    /// Returns the GID that the process that triggered this request runs under.
224    fn gid(&self) -> u32;
225
226    /// Returns the PID of the process that triggered this request.
227    fn pid(&self) -> u32;
228
229    /// Create an error response for this Request
230    fn reply_err(&self, errno: Errno) -> Response<'_> {
231        Response::new_error(errno)
232    }
233}

255                self.header.gid
256            }
257
258            #[inline]
259            fn pid(&self) -> u32 {
260                self.header.pid
261            }
262        }
263    };

1207    /// If a process issuing a FUSE filesystem request is interrupted, the
1208    /// following will happen:
1209    ///
1210    ///   1) If the request is not yet sent to userspace AND the signal is
1211    ///      fatal (SIGKILL or unhandled fatal signal), then the request is
1212    ///      dequeued and returns immediately.
1213    ///
1214    ///   2) If the request is not yet sent to userspace AND the signal is not
1215    ///      fatal, then an 'interrupted' flag is set for the request.  When
1216    ///      the request has been successfully transferred to userspace and
1217    ///      this flag is set, an INTERRUPT request is queued.
1218    ///
1219    ///   3) If the request is already sent to userspace, then an INTERRUPT
1220    ///      request is queued.
1221    ///
1222    /// [Interrupt] requests take precedence over other requests, so the
1223    /// userspace filesystem will receive queued [Interrupt]s before any others.
1224    ///
1225    /// The userspace filesystem may ignore the [Interrupt] requests entirely,
1226    /// or may honor them by sending a reply to the **original** request, with
1227    /// the error set to [Errno::EINTR].
1228    ///
1229    /// It is also possible that there's a race between processing the
1230    /// original request and its [Interrupt] request.  There are two
1231    /// possibilities:
1232    ///
1233    /// 1. The [Interrupt] request is processed before the original request is
1234    ///    processed
1235    ///
1236    /// 2. The [Interrupt] request is processed after the original request has
1237    ///    been answered
1238    ///
1239    /// If the filesystem cannot find the original request, it should wait for
1240    /// some timeout and/or a number of new requests to arrive, after which it
1241    /// should reply to the [Interrupt] request with an [Errno::EAGAIN] error.
1242    /// In case (1) the [Interrupt] request will be requeued.  In case (2) the
1243    /// [Interrupt] reply will be ignored.
1244    #[derive(Debug)]
1245    pub struct Interrupt<'a> {
1246        header: &'a fuse_in_header,
1247        arg: &'a fuse_interrupt_in,
1248    }

2129#[derive(Debug)]
2130pub struct AnyRequest<'a> {
2131    header: &'a fuse_in_header,
2132    data: &'a [u8],
2133}
2134impl_request!(AnyRequest<'_>);

2136impl<'a> AnyRequest<'a> {
2137    pub fn operation(&self) -> Result<Operation<'a>, RequestError> {
2138        // Parse/check opcode
2139        let opcode = fuse_opcode::try_from(self.header.opcode)
2140            .map_err(|_: InvalidOpcodeError| RequestError::UnknownOperation(self.header.opcode))?;
2141        // Parse/check operation arguments
2142        op::parse(self.header, &opcode, self.data).ok_or(RequestError::InsufficientData)
2143    }
2144}

2146impl<'a> fmt::Display for AnyRequest<'a> {
2147    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
2148        if let Ok(op) = self.operation() {
2149            write!(
2150                f,
2151                "FUSE({:3}) ino {:#018x} {}",
2152                self.header.unique, self.header.nodeid, op
2153            )
2154        } else {
2155            write!(
2156                f,
2157                "FUSE({:3}) ino {:#018x}",
2158                self.header.unique, self.header.nodeid
2159            )
2160        }
2161    }

2164impl<'a> TryFrom<&'a [u8]> for AnyRequest<'a> {
2165    type Error = RequestError;
2166
2167    fn try_from(data: &'a [u8]) -> Result<Self, Self::Error> {
2168        // Parse a raw packet as sent by the kernel driver into typed data. Every request always
2169        // begins with a `fuse_in_header` struct followed by arguments depending on the opcode.
2170        let data_len = data.len();
2171        let mut arg_iter = ArgumentIterator::new(data);
2172        // Parse header
2173        let header: &fuse_in_header = arg_iter
2174            .fetch()
2175            .ok_or_else(|| RequestError::ShortReadHeader(arg_iter.len()))?;
2176        // Check data size
2177        if data_len < header.len as usize {
2178            return Err(RequestError::ShortRead(data_len, header.len as usize));
2179        }
2180        Ok(Self {
2181            header,
2182            data: &data[mem::size_of::<fuse_in_header>()..header.len as usize],
2183        })
2184    }
2185}
